Actor NewStrifeBishop : SamsaraMonsterRemixStrifeBishop
{
	Tag "Bishop"
	Species "StrifeEnemies"
	Health 500
	ProjectilePassHeight 64
	-BOSS
	-BOSSDEATH
	+DONTRIP
	+NOINFIGHTING
	DropItem "StrifeBishopDropMixer", 128
	States
	{
		Spawn:
			MLDR A 0
			MLDR A 1 A_JumpIfInventory("AlreadySpawned",1,"Spawn2")
			MLDR A 1 ACS_NamedExecuteAlways("ReMixerSpawnScript",0,1,0)
			MLDR A 0 A_GiveInventory("AlreadySpawned",1)
		Spawn2:
			MLDR A 10 A_Look
			Loop
		See:
			MLDR AABBCCDD 3 A_Chase
			Loop
		Missile:
			MLDR E 3 A_FaceTarget
			MLDR F 2 Bright A_CustomMissile("NewBishopMissile", 56, 0, 0, CMF_AIMOFFSET)
			Goto See
		Pain:
			MLDR D 1 A_Pain
			Goto See
		Death:
			TNT1 A 0 ACS_NamedExecuteAlways("Samsara_KillCount", 0, 0)
			MLDR G 3 Bright
			MLDR H 5 Bright A_Scream
			MLDR I 4 Bright A_TossGib
			MLDR J 4 Bright A_Explode(64,64,1,1)
			MLDR KL 3 Bright
			MLDR M 4 Bright A_NoBlocking
			MLDR N 4 Bright
			MLDR O 4 Bright A_TossGib
			MLDR P 4 Bright
			MLDR Q 4 Bright A_TossGib
			MLDR R 4 Bright
			MLDR S 4 Bright A_TossGib
			MLDR T 4 Bright
			MLDR U 4 Bright A_TossGib
			MLDR V 4 Bright
			MLDR V 1
			MLDR V 10
			MLDR V -1 A_KillMaster
			Stop
		Pain.MightyBoot:
			TNT1 A 0 A_ChangeFlag("NOPAIN",1)
			TNT1 A 0 A_PlaySound("duke/kickhit", CHAN_BODY)
			MLDR D 2 A_FaceTarget
			TNT1 A 0 A_Recoil(14)
			MLDR D 16 A_Pain
			Goto See

		// SamsaraMonsterRemixStrifeBishop states
		Disintegrate:
		XDeath.PunchDagger:
		XDeath.ElectricBolt:
		XDeath.PoisonBolt:
		XDeath.StrifeExplosive:
		XDeath.StrifeSigil:
			TNT1 A -1 A_Jump(256,"Death")
			Wait
	}
}
