	DeathSound "vdr/nshdth"

	+FLOORCLIP
	+FLOAT
	+NOGRAVITY
	+NOBLOOD
	Tag "Nightshade"
	Species "SamsaraNightshade"
	DamageFactor "LostSoul64Explosion", 0
	States
	{
		Spawn:
			VD65 A 0
			VD65 A 1 A_JumpIfInventory("AlreadySpawned",1,"Spawn2")
			VD65 A 1 ACS_NamedExecuteAlways("ReMixerSpawnScript",0,1,0)
			VD65 A 0 A_GiveInventory("AlreadySpawned",1)
			VD65 A 0
		Spawn2:
			VD65 AB 10 A_Look
			loop
		See:
			TNT1 A 0 A_SetTranslucent(0.75,0)
			VD65 AABBCCDD 3 A_Chase
			goto See+1
		Missile:
			TNT1 A 0 A_ChangeFlag("NOPAIN",0)
			TNT1 A 0 A_SetTranslucent(1.0,0)
			VD65 EE 4 Bright A_FaceTarget
			TNT1 A 0 A_FaceTarget
			VD65 F 8 Bright A_CustomMissile("MMRandomizer_ShadeMissile",32)
			VD65 B 4 A_SetTranslucent(0.87,0)
			goto See
		Pain.MightyBoot:
			TNT1 A 0 A_ChangeFlag("NOPAIN",1)
			TNT1 A 0 A_SetTranslucent(1.0,0)
			TNT1 A 0 A_PlaySound("duke/kickhit",CHAN_BODY)
			VD65 G 2 A_FaceTarget
			TNT1 A 0 A_Recoil(14)
			VD65 G 16 A_Pain
			Goto See
		Pain:
			VD65 G 2 A_SetTranslucent(1.0,0)
			VD65 G 2 A_Pain
			goto See
		Pain.DoomDamageTypeStunner:
			TNT1 A 0 A_SetTranslucent(1.0,0)
			VD65 G 2 A_ChangeFlag("NOPAIN",true)
			VD65 G 300 A_Pain
			TNT1 A 0 A_ChangeFlag("NOPAIN",false)
			goto See
		Death:
			TNT1 A 0 ACS_NamedExecuteAlways("Samsara_KillCount", 0, 0)
			VD65 H 4 A_SetTranslucent(1.0,0)
			VD65 I 4 A_SetTranslucent(0.9,0)
			TNT1 A 0 A_Scream
			VD65 J 4 A_SetTranslucent(0.8,0)
			TNT1 A 0 A_NoBlocking
			VD65 K 4 A_SetTranslucent(0.7,0)
			VD65 L 4 A_SetTranslucent(0.6,0)
			VD65 M 4 A_SetTranslucent(0.5,0)
			VD65 N 4 A_SetTranslucent(0.4,0)
			VD65 O 4 A_SetTranslucent(0.3,0)
			VD65 P 4 A_SetTranslucent(0.2,0)
			stop
	}
}
