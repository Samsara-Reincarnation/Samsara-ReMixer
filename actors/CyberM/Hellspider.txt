Actor CMHellspider : SamsaraMonsterBeastTotalGib
{
    obituary "%o was overwhelmed by a Hellspider."
    hitobituary "%o was sliced up by a Hellspider."
    scale 0.45
    Health 300
    Radius 28
    Height 64
	Mass 900
    painchance 85
    meleedamage 4
	Species "CMEnemies"	
	Damage 1
	speed 8
	Tag "Hellspider"
	seesound "cmhspider/sight"
    activesound "cmhspider/active"
	painsound "cmhspider/pain"
	deathsound "cmhspider/death"
	PainChance "MightyBoot", 255
	PainChance "MarathonStun", 255
	Bloodcolor Green
	Meleerange 72
	+FLOORCLIP
	+BOSSDEATH	
	+DONTRIP	
	+NOINFIGHTING
    Monster
    States
    {
    Spawn:
		C_HS B 0
		C_HS B 1 A_JumpIfInventory("AlreadySpawned",1,"Spawn2")
		C_HS B 1 ACS_NamedExecuteAlways("ReMixerSpawnScript",0,1,0)
		C_HS B 0 A_GiveInventory("AlreadySpawned",1)
	Spawn2:
		C_HS BB 4 A_Look
		loop
    See:
		C_HS A 2 A_Chase
		C_HS A 2 A_Chase
		C_HS B 2 A_Chase
		TNT1 A 0 A_JumpIfCloser(180,"Lean")			
		C_HS B 2 A_Chase
		C_HS C 2 A_Chase	
		C_HS C 2 A_Chase
		C_HS D 2 A_Chase
		TNT1 A 0 A_JumpIfCloser(180,"Lean")			
		C_HS D 2 A_Chase
		Goto See
    Lean:
		C_HS EE 3 A_FaceTarget
		C_HS FF 3 A_FaceTarget
		TNT1 A 0 A_PlaySound("cmhspider/lean")
		TNT1 A 0 A_SkullAttack
		C_HS A 0 ThrustThingZ(0,25,0,0)
		C_HS F 2 A_Recoil(-12)
		TNT1 A 0 A_FaceTarget
	LeanLoop:
		C_HS F 3 A_CheckFloor("See")	
		Goto LeanLoop
	Melee:
		TNT1 A 0 A_ChangeFlag("NOPAIN",0)
		C_HS EEE 2 A_FaceTarget
		C_HS FF 2 A_FaceTarget
		C_HS F 1 A_FaceTarget
		TNT1 A 0 A_PlaySound("cmhspider/lean")
		C_HS G 6 A_CustomMeleeAttack(Random(12,32),"","")
		C_HS HCCEE 3 A_FaceTarget
		Goto See
	Pain:
		C_HS I 2 A_Pain
		C_HS I 2
		Goto See
    Death:
		TNT1 A 0 ACS_NamedExecuteAlways("Samsara_KillCount", 0, 0)
		TNT1 A 0 A_SpawnItemEx("UniversalBloodyHellGoreDeathEffect",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION|SXF_USEBLOODCOLOR)
		C_HS I 2 A_Die
		C_HS I 1 A_Noblocking
		C_HS J 2 A_Scream	
		C_HS J 6
		C_HS K 3
		C_HS K -1 A_KillMaster
		stop
	Pain.MightyBoot:
		TNT1 A 0 A_ChangeFlag("NOPAIN",1)
		TNT1 A 0 A_PlaySoundEx("duke/kickhit","body",0,0)
		C_HS I 2 A_FaceTarget
		TNT1 A 0 A_Recoil(14)
		C_HS I 16 A_Pain
		Goto See
	Pain.ShrinkerDMG:
		C_HS I 0
		"####" "#" 0 A_SetUserVar("user_scalex", scalex*65536.0)
		"####" "#" 0 A_SetUserVar("user_scaley", scaley*65536.0)
		"####" "#" 0 A_ChangeFlag("NOPAIN", 1)
		"####" "#" 0 A_ChangeFlag("NODAMAGE", 1)
		"####" "#" 0 A_NoBlocking
		"####" "#" 0 A_PlaySound("duke/shrink")
		"####" "#" 1 A_SetScale(0.95*(user_scalex/65536.0),0.95*(user_scaley/65536.0))
		"####" "#" 1 A_SetScale(0.9*(user_scalex/65536.0),0.9*(user_scaley/65536.0))
		"####" "#" 0 A_SpawnItemEx("CMHellspider_ShrinkEffect",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION)
		"####" "#" 1 A_SetScale(0.85*(user_scalex/65536.0),0.85*(user_scaley/65536.0))
		"####" "#" 1 A_SetScale(0.8*(user_scalex/65536.0),0.8*(user_scaley/65536.0))
		"####" "#" 0 A_SpawnItemEx("CMHellspider_ShrinkEffect",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION)
		"####" "#" 1 A_SetScale(0.75*(user_scalex/65536.0),0.75*(user_scaley/65536.0))
		"####" "#" 1 A_SetScale(0.7*(user_scalex/65536.0),0.7*(user_scaley/65536.0))
		"####" "#" 0 A_SpawnItemEx("CMHellspider_ShrinkEffect",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION)
		"####" "#" 1 A_SetScale(0.65*(user_scalex/65536.0),0.65*(user_scaley/65536.0))
		"####" "#" 1 A_SetScale(0.6*(user_scalex/65536.0),0.6*(user_scaley/65536.0))
		"####" "#" 0 A_SpawnItemEx("CMHellspider_ShrinkEffect",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION)
		"####" "#" 1 A_SetScale(0.55*(user_scalex/65536.0),0.55*(user_scaley/65536.0))
		"####" "#" 1 A_SetScale(0.5*(user_scalex/65536.0),0.5*(user_scaley/65536.0))
		"####" "#" 0 A_SpawnItemEx("CMHellspider_ShrinkEffect",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION)
		"####" "#" 1 A_SetScale(0.45*(user_scalex/65536.0),0.45*(user_scaley/65536.0))
		"####" "#" 1 A_SetScale(0.4*(user_scalex/65536.0),0.4*(user_scaley/65536.0))
		"####" "#" 0 A_SpawnItemEx("CMHellspider_ShrinkEffect",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION)
		"####" "#" 1 A_SetScale(0.35*(user_scalex/65536.0),0.35*(user_scaley/65536.0))
		"####" "#" 1 A_SetScale(0.3*(user_scalex/65536.0),0.3*(user_scaley/65536.0))
		"####" "#" 0 A_SpawnItemEx("CMHellspider_ShrinkEffect",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION)
		"####" "#" 1 A_SetScale(0.25*(user_scalex/65536.0),0.25*(user_scaley/65536.0))
		"####" "#" 4 A_SetScale(0.2*(user_scalex/65536.0),0.2*(user_scaley/65536.0))
		"####" "#" 0 A_SpawnItemEx("CMHellspider_Shrink", 0, 0, 0, 0, 0, 0, 0, SXF_NOCHECKPOSITION|SXF_TRANSFERPOINTERS|SXF_TRANSFERSPECIAL|SXF_TRANSFERAMBUSHFLAG|SXF_SETMASTER,0,tid)
		TNT1 A 2
		TNT1 A 0 A_ChangeFlag("NODAMAGE", 1)
		TNT1 A 0 A_ChangeFlag("NOPAIN", 1)
		TNT1 A 0 A_ChangeFlag("VULNERABLE", 1)
		TNT1 A 0 A_ChangeFlag("SHOOTABLE", 0)
		Goto ShrunkLoop
	ShrunkLoop:
		TNT1 A 2 A_JumpIfInventory("MasterDeadToken",1,"ShrunkDie")
		loop
	ShrunkDie:
		TNT1 A 0
		TNT1 A 1 A_Die("Shrink")
		stop
	Death.Shrink:
		TNT1 A 0
		TNT1 A 1 A_KillMaster
		stop
	}
}
