actor Goldfinger_MP40EnemyDrop : SamsaraSlotTwoDrop
{
    Gravity 0.33
    +DROPPED
    +NEVERRESPAWN
    States
    {
      Spawn:
        GF00 Z 0 NoDelay ACS_NamedExecuteWithResult("Samsara_SwitchableObject", 35)
        goto SpawnGameModeCheck

      SpawnActual:
        GF00 Z 0 A_ChangeFlag("THRUACTORS",false)
        GF00 Z 0 A_ChangeFlag("NOTIMEFREEZE",false)
      SpawnActualAnim:
        GF00 Z 0 ThrustThing(random(0,256)/360,random(1,2),0,0)
        GF00 Z 0 ThrustThingZ(0,random(0,2),0,0)
        GF00 Z 0 A_SetAngle(random(0,360))
        GF00 Z 0 A_SetPitch(random(0,360))
      SpawnActualAnimLoop:
        GF00 Z 0 A_SetPitch(pitch-15)
        GF00 Z 0 A_SetAngle(angle-15)
        GF00 Z 1 A_CheckFloor("SpawnActualAnimLand")
        loop

      SpawnActualAnimLand:
        GF00 Z 0 A_PlaySound("Goldeneye/DropLand")
        GF00 Z 0 A_Stop
        GF00 Z -1 A_SetPitch(0)
        stop
    }
}
