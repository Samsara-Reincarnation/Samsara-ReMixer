Actor MMRandomizer_TriteHole
{
	Height 32
	Mass 25
	Radius 16
	Species "Trite"
	Scale 0.55
	+ISMONSTER
	-COUNTKILL
	+NODAMAGE
	+THRUSPECIES
	-SHOOTABLE
	+SOLID
	var int user_triteamount;
	var int user_loopcounter;
	States
	{
		Spawn:
			VD39 LL 0 A_SetUserVar("user_triteamount",5)
			VD39 L 5 A_Look
			Loop
		See:
			VD39 L 35
		SeeLoop:
			VD39 L 0 A_JumpIf(user_loopcounter >= user_triteamount, "Remove")
			VD39 L 0 A_SetUserVar("user_loopcounter",user_loopcounter+1)
			VD39 MN 5
			VD39 N 0 A_SpawnItemEx("MMRandomizer_Trite",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION|SXF_TRANSFERSPECIAL|SXF_TRANSFERPOINTERS,0,tid)
			VD39 L 35
			Loop
		Remove:
			VD39 LLLLLLLLLLLLLLLLLLLL 4 A_Fadeout(0.05)
			Stop
	}
}
