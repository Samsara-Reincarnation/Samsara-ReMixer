Actor ChokeMe : Weapon
{
    Weapon.SelectionOrder 1
	Inventory.Amount 1
	Inventory.MaxAmount 1	
	+INVENTORY.UNDROPPABLE
	+WEAPON.NOALERT
	+AUTOACTIVATE	
	+ALWAYSPICKUP
	states
	{
	Ready:
		TNT1 A 0 A_JumpIfInventory("AmAlreadyChoked",1,"ReadyLoop")
		TNT1 A 0 A_GiveInventory("AmAlreadyChoked",1)	
		TNT1 A 0 A_Print("Spam the Alt Fire Button to take it off!")
	ReadyLoop:
		TNT1 A 0 A_Explode(5, 1)
//		7H9X A 0 A_PlaySound("Hand/Choke",CHAN_WEAPON)
		7H9X A 1 A_WeaponReady(WRF_NOSWITCH)
		7H9X ABCDEDCBAFGF 3 A_WeaponReady(WRF_NOSWITCH|WRF_NOFIRE)
		Loop
    Select:
//    	TNT1 A 0 A_SetDamageType("chokeEM")		
		TNT1 A 0 A_WeaponReady(WRF_NOSWITCH|WRF_NOFIRE)
	SelectLoop:	
		7H9X ABCDEFGABCDEFG 3 A_Raise
		Loop
	Deselect:
		TNT1 A 0 A_TakeInventory("AmAlreadyChoked",1)
		TNT1 A 0 A_SpawnItemEx("Chokinghand_2",0,0,0,0,0,0,SXF_NOCHECKPOSITION)
	    TNT1 A 0 A_TakeInventory("Chokeme",1)	
		7H9X ABCDEFGABCDEFG 3 A_Lower
		Stop
    Fire:
		TNT1 A 0 A_Explode(1, 1)
//		7H9X A 0 A_PlaySound("Hand/Choke",TRUE)
		Goto ReadyLoop
	AltFire:
		TNT1 A 1 A_FireCustomMissile("Flying_BloodHand",0)
		TNT1 A 1 A_TakeInventory("AmAlreadyChoked",1)		
	    TNT1 A 1 A_TakeInventory("Chokeme",1)
		Stop
	}
}
