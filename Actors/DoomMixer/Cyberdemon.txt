ACTOR CyberDemon3 : CyberDemon2
{
	SeeSound "doom/cyber/sight"
	PainSound "doom/cyber/pain"
	DeathSound "doom/cyber/death"
	ActiveSound "doom/cyber/active"
	States
	{
	Spawn:
		CYB2 A 0
		CYB2 A 1 A_JumpIfInventory("AlreadySpawned",1,"Spawn2")
		CYB2 A 1 ACS_NamedExecuteAlways("ReMixerSpawnScript",0,1,0)
		CYB2 A 0 A_GiveInventory("AlreadySpawned",1)
		CYB2 A 0
	Spawn2:
		CYB2 AB 10 A_Look
		Loop
	See:
        TNT1 A 0 A_JumpIfInventory("DukeBossSee",1,"See2")
        TNT1 A 0 A_JumpIfInventory("CalebBossSee",1,"See2")		
        TNT1 A 0 A_JumpIfInventory("EleenaBossSee",1,"See2")		
        TNT1 A 0 A_JumpIfInventory("LoWangBossSee",1,"See2")	
        TNT1 A 0 A_JumpIfInventory("LeonardBossSee",1,"See2")						
        TNT1 A 0 A_JumpIfInventory("HLBossSee",1,"See2")
		TNT1 A 0 A_GiveInventory("DukeBossSee",1)
        TNT1 A 0 A_GiveInventory("CalebBossSee",1)		
        TNT1 A 0 A_GiveInventory("EleenaBossSee",1)	
        TNT1 A 0 A_GiveInventory("LoWangBossSee",1)
        TNT1 A 0 A_GiveInventory("LeonardBossSee",1)
		TNT1 A 0 A_GiveInventory("HLBossSee",1)
        TNT1 A 0 A_GiveToTarget("DukeBossSightTaunt")
        TNT1 A 0 A_GiveToTarget("CalebBossSightTaunt")
        TNT1 A 0 A_GiveToTarget("EleenaBossSightTaunt")
        TNT1 A 0 A_GiveToTarget("LoWangBossSightTaunt")
        TNT1 A 0 A_GiveToTarget("LeonardBossSightTaunt")		
		TNT1 A 0 A_GiveToTarget("HLBossSightMusicQueue")
    See2:
        CYB2 A 3 A_PlaySound("doom/cyber/hoof", CHAN_BODY)
		CYB2 ABBCC 3 A_Chase
		CYB2 D 3 A_PlaySound("doom/spider/walk", CHAN_BODY)
		CYB2 D 3 A_Chase
        loop
	Missile:
		CYB2 E 6 A_FaceTarget
		CYB2 F 12 A_CustomMissile("Rocket_Samsara",32,0,0)
		CYB2 E 12 A_FaceTarget
		CYB2 F 12 A_CustomMissile("Rocket_Samsara",32,0,0)
		CYB2 E 12 A_FaceTarget
		CYB2 F 12 A_CustomMissile("Rocket_Samsara",32,0,0)
		Goto See2
	Pain:
		CYB2 G 10 A_Pain
		Goto See2
	Death:
        TNT1 A 0 ACS_NamedExecuteAlways("Samsara_KillCount", 0, 0)
		CYB2 H 10
		CYB2 I 10 A_Scream
		TNT1 A 0 A_SpawnItemEx("UniversalBloodyHellGoreXDeathEffect",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION|SXF_USEBLOODCOLOR)	
		CYB2 JKL 10	
		CYB2 M 10 A_NoBlocking
		CYB2 NO 10
		CYB2 P 30
		CYB2 P -1 ACS_NamedExecuteWithResult("SamsaraBossDeathToggler")
		Stop
	}
}

ACTOR Rocket_Samsara
{
  Radius 11
  Height 8
  Speed 20
  Damage 20
  Projectile
  +RANDOMIZE
  +DEHEXPLOSION
  +ROCKETTRAIL
  SeeSound "doom/weapons/rocklf"
  DeathSound "doom/weapons/rocklx"
  Obituary "$OB_MPROCKET" // "%o rode %k's rocket."
  States
  {
  Spawn:
    MIS2 A 1 Bright
    Loop
  Death:
    MIS2 B 8 Bright A_Explode
    MIS2 C 6 Bright
    MIS2 D 4 Bright
    Stop
  }
}
